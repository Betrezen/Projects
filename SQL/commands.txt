The simple ways how we can find out keys which don’t have the product info.
SELECT token_id FROM NIM_NBI_TOKEN minus SELECT NIM_NBI_PRODUCT.TOKEN_ID FROM NIM_NBI_PRODUCT;
SELECT * FROM NIM_NBI_TOKEN WHERE NIM_NBI_TOKEN.TOKEN_ID NOT IN (SELECT NIM_NBI_PRODUCT.TOKEN_ID FROM NIM_NBI_PRODUCT);
SELECT * FROM NIM_NBI_TOKEN t1 WHERE NOT EXISTS (SELECT 1 FROM NIM_NBI_PRODUCT t2 WHERE t1.token_id=t2.token_id);
